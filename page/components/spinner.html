PHP

use yh\mdc\components\Spinner;
 
echo Spinner::one(['size' => Spinner::EXSMALL])
    ->setId('spinner1')
    ->render();
echo '<br>';
echo Spinner::one(['size' => Spinner::SMALL])
    ->setId('spinner2')
    ->render();
echo '<br>';
echo Spinner::one(['size' => Spinner::MEDIUM, 'enabled' => true])
    ->setId('spinner3')
    ->render();
echo '<br>';
echo Spinner::one(['size' => Spinner::LARGE, 'enabled' => true])
    ->setId('spinner4')
    ->render();

JavaScript

app.controls.add('spinner1', 'spinner', []);
app.controls.add('spinner2', 'spinner', []);
app.controls.add('spinner3', 'spinner', []);
app.controls.add('spinner4', 'spinner', []);
 
app.controls.item("spinner1").show();
app.controls.item("spinner2").show();

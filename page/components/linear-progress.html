Linear progress
@see PHP class

PHP

use yh\mdc\components\LinearProgress;
 
echo LinearProgress::one('Progress')
    ->setProperty([
        'closed' => false,
        'indeterminate' => true
    ])
    ->setOptions([
        'style' => 'margin-bottom: 10px'
    ])
    ->setId('progress1')
    ->render();
 
echo LinearProgress::one('Progress')
    ->setProperty([
        'closed' => false,
        'progress' => 0.3,
        'buffer' => 0.8
    ])
    ->setOptions([
        'style' => 'margin-bottom: 10px'
    ])
    ->setId('progress2')
    ->render();

JavaScript

app.controls.add('progress1','linearProgress',[]);
app.controls.add('progress2','linearProgress',{"buffer":0.8,"progress":0.3});
app.controls.item('progress2').progress = 0.5;
app.controls.item('progress2').buffer = 0.8;

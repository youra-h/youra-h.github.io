let TextField2 = app.controls.item("TextField2");
TextField2.minLength = 5;
TextField2.trailingIcon.click(function () {
    if (this.parent.type === "text") {
        this.parent.type = "password";
    } else {
        this.parent.type = "text";
    }
    this.parent.focus();
}, TextField2.trailingIcon);

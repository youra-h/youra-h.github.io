PHP

use yh\mdc\components\IconButton;
 
echo IconButton::one('settings')
    ->setProperty([
        'icon' => 'settings'
    ])
    ->setOptions([
        'style' => ['margin-right' => '10px']
    ])
    ->setId('icon1')
    ->render();
 
echo IconButton::one()
    ->setProperty([
        'icon' => 'visibility',
        'iconOn' => 'visibility_off',
        'toggle' => true
    ])
    ->setOptions([
        'style' => ['margin-right' => '10px']
    ])
    ->setId('icon2')
    ->render();

JavaScript

app.controls.add('icon1', 'iconButton', []);
app.controls.add('icon2', 'iconButton', []).on = true;
